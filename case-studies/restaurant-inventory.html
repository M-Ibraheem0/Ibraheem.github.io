Data Integration

# Standardized inventory consolidation
import pandas as pd
from datetime import datetime

def standardize_inventory(pos_data, location_id):
    # Normalize item names and units
    standardized = pos_data.copy()
    standardized['location_id'] = location_id
    standardized['item_name'] = standardized['item_name'].str.upper().str.strip()
    standardized['unit_cost'] = standardized['total_cost'] / standardized['quantity']
    
    return standardized

# Automated reorder point calculation
def calculate_reorder_point(usage_history, lead_time_days=3):
    daily_avg = usage_history.groupby('date')['quantity_used'].mean()
    safety_stock = daily_avg.std() * 1.5  # 1.5 sigma safety
    reorder_point = (daily_avg.mean() * lead_time_days) + safety_stock
    
    return max(reorder_point, 0)

Waste Tracking Algorithm

# Predictive waste reduction
def analyze_waste_patterns(inventory_data):
    waste_by_item = inventory_data.groupby(['item_name', 'location_id']).agg({
        'waste_amount': 'sum',
        'total_purchased': 'sum',
        'days_to_expiry': 'mean'
    })
    
    waste_by_item['waste_percentage'] = (
        waste_by_item['waste_amount'] / waste_by_item['total_purchased'] * 100
    )
    
    # Flag high-waste items for optimization
    high_waste_items = waste_by_item[waste_by_item['waste_percentage'] > 15]
    
    return high_waste_items

# Automated ordering optimization
def optimize_order_quantity(item_usage, shelf_life, current_inventory):
    daily_usage = item_usage.mean()
    optimal_order = daily_usage * min(shelf_life * 0.8, 14)  # Max 14 days
    
    return max(optimal_order - current_inventory, 0)
