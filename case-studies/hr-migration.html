The Challenge

Legacy System Data Prison

The company faced a critical migration challenge with significant business risks:

  • Legacy Complexity: 15-year-old AS/400 system with undocumented data structures and relationships
  • Data Volume: 2.3 million records across 47 interconnected tables with complex hierarchies
  • Compliance Risk: GDPR, SOX, and labor law requirements for data accuracy and audit trails
  • Business Continuity: Zero tolerance for downtime during payroll and benefits processing
  • Data Quality Issues: Inconsistent formats, duplicate records, and orphaned data relationships

import pandas as pd
from datetime import datetime
import re

def standardize_employee_data(legacy_df):
    # Standardize date formats
    def parse_date(date_str):
        formats = ['%Y%m%d', '%m/%d/%y', '%d-%b-%Y']
        for fmt in formats:
            try:
                return datetime.strptime(str(date_str), fmt)
            except:
                continue
        return None
    
    # Clean and standardize names
    legacy_df['full_name'] = legacy_df['first_name'].str.title() + ' ' + \
                            legacy_df['last_name'].str.title()
    
    # Standardize employee IDs
    legacy_df['employee_id'] = legacy_df['emp_id'].apply(
        lambda x: f"EMP{str(x).zfill(6)}"
    )
    
    return legacy_df

# Data validation pipeline
def validate_migration_data(source_df, target_df):
    validation_results = {
        'record_count_match': len(source_df) == len(target_df),
        'key_integrity': set(source_df['employee_id']) == set(target_df['employee_id']),
        'data_completeness': target_df.isnull().sum().sum() == 0
    }
    
    return validation_results

Migration Pipeline

# Robust migration with rollback capability
class HRMigrationPipeline:
    def __init__(self, source_conn, target_conn):
        self.source = source_conn
        self.target = target_conn
        self.backup_tables = []
    
    def create_backup(self, table_name):
        backup_name = f"{table_name}_backup_{datetime.now().strftime('%Y%m%d')}"
        self.target.execute(f"CREATE TABLE {backup_name} AS SELECT * FROM {table_name}")
        self.backup_tables.append(backup_name)
    
    def migrate_table(self, source_table, target_table, batch_size=1000):
        try:
            # Create backup before migration
            self.create_backup(target_table)
            
            # Extract data in batches
            total_records = self.source.execute(
                f"SELECT COUNT(*) FROM {source_table}"
            ).fetchone()[0]
            
            for offset in range(0, total_records, batch_size):
                batch_data = pd.read_sql(
                    f"SELECT * FROM {source_table} LIMIT {batch_size} OFFSET {offset}",
                    self.source
                )
                
                # Transform and validate
                transformed_data = self.transform_data(batch_data)
                validation_passed = self.validate_batch(transformed_data)
                
                if validation_passed:
                    transformed_data.to_sql(target_table, self.target, 
                                          if_exists='append', index=False)
                else:
                    raise Exception(f"Validation failed for batch at offset {offset}")
            
            return True
            
        except Exception as e:
            self.rollback_migration()
            raise e
