# Comprehensive retail analytics platform
import pandas as pd
import numpy as np
from datetime import datetime, timedelta
from sklearn.cluster import KMeans
from sklearn.ensemble import RandomForestRegressor

class RetailAnalyticsEngine:
    def __init__(self):
        self.demand_model = RandomForestRegressor(n_estimators=100, random_state=42)
        self.customer_segmenter = KMeans(n_clusters=5, random_state=42)
        self.price_elasticity = {}
        
    def analyze_product_performance(self, sales_data, inventory_data):
        """Comprehensive product performance analysis"""
        # Merge sales and inventory data
        product_analysis = sales_data.groupby(['product_id', 'store_id']).agg({
            'quantity_sold': 'sum',
            'revenue': 'sum',
            'transaction_count': 'count',
            'discount_amount': 'sum'
        }).reset_index()
        
        # Add inventory metrics
        inventory_metrics = inventory_data.groupby(['product_id', 'store_id']).agg({
            'current_stock': 'last',
            'stockout_days': 'sum',
            'overstock_days': 'sum',
            'turnover_rate': 'mean'
        }).reset_index()
        
        analysis = product_analysis.merge(inventory_metrics, on=['product_id', 'store_id'])
        
        # Calculate key performance indicators
        analysis['revenue_per_unit'] = analysis['revenue'] / analysis['quantity_sold']
        analysis['sell_through_rate'] = analysis['quantity_sold'] / (analysis['quantity_sold'] + analysis['current_stock'])
        analysis['stockout_impact'] = analysis['stockout_days'] * analysis['revenue_per_unit'] * 10  # Estimated daily demand
        analysis['overstock_cost'] = analysis['overstock_days'] * analysis['revenue_per_unit'] * 0.02  # Carrying cost
        
        # Performance scoring
        analysis['performance_score'] = (
            (analysis['sell_through_rate'] * 0.3) +
            (analysis['turnover_rate'] / analysis['turnover_rate'].max() * 0.3) +
            ((1 - analysis['stockout_days'] / 365) * 0.2) +
            ((1 - analysis['overstock_days'] / 365) * 0.2)
        ) * 100
        
        return analysis
    
    def optimize_pricing(self, product_id, store_id, sales_history, competitor_prices=None):
        """Dynamic pricing optimization based on demand elasticity"""
        product_sales = sales_history[
            (sales_history['product_id'] == product_id) & 
            (sales_history['store_id'] == store_id)
        ].copy()
        
        if len(product_sales) < 30:  # Need sufficient data
            return {'recommended_price': None, 'confidence': 'LOW'}
            
        # Calculate price elasticity
        product_sales['price_change'] = product_sales['unit_price'].pct_change()
        product_sales['quantity_change'] = product_sales['quantity_sold'].pct_change()
        
        # Remove outliers and invalid data
        valid_data = product_sales[
            (abs(product_sales['price_change']) < 0.5) &  # Less than 50% price change
            (abs(product_sales['quantity_change']) < 2.0) &  # Less than 200% quantity change
            (product_sales['price_change'] != 0)
        ]
        
        if len(valid_data) < 10:
            return {'recommended_price': None, 'confidence': 'LOW'}
            
        # Calculate elasticity
        elasticity = (valid_data['quantity_change'] / valid_data['price_change']).mean()
        
        # Current metrics
        current_price = product_sales['unit_price'].iloc[-1]
        current_quantity = product_sales['quantity_sold'].iloc[-7:].mean()  # Weekly average
        current_margin = product_sales['margin_percent'].iloc[-1]
        
        # Optimize price for maximum profit
        price_scenarios = []
        for price_change in np.arange(-0.2, 0.21, 0.05):  # -20% to +20% in 5% steps
            new_price = current_price * (1 + price_change)
            predicted_quantity = current_quantity * (1 + elasticity * price_change)
            predicted_revenue = new_price * predicted_quantity
            predicted_margin = (new_price - product_sales['cost_per_unit'].iloc[-1]) / new_price
            predicted_profit = predicted_revenue * predicted_margin
            
            price_scenarios.append({
                'price': new_price,
                'quantity': predicted_quantity,
                'revenue': predicted_revenue,
                'profit': predicted_profit,
                'margin': predicted_margin
            })
        
        # Find optimal price
        optimal_scenario = max(price_scenarios, key=lambda x: x['profit'])
        
        # Consider competitor pricing if available
        if competitor_prices:
            avg_competitor_price = np.mean(competitor_prices)
            if optimal_scenario['price'] > avg_competitor_price * 1.15:  # More than 15% above competitors
                optimal_scenario['price'] = avg_competitor_price * 1.10  # Price 10% above average
                
        return {
            'recommended_price': round(optimal_scenario['price'], 2),
            'expected_quantity': round(optimal_scenario['quantity']),
            'expected_revenue': round(optimal_scenario['revenue'], 2),
            'expected_profit': round(optimal_scenario['profit'], 2),
            'price_elasticity': round(elasticity, 3),
            'confidence': 'HIGH' if len(valid_data) > 20 else 'MEDIUM'
        }

# Advanced customer analytics and segmentation
from sklearn.preprocessing import StandardScaler
from datetime import datetime, timedelta

class CustomerAnalytics:
    def __init__(self):
        self.scaler = StandardScaler()
        self.segment_profiles = {}
        
    def create_customer_features(self, customer_data, transaction_data):
        """Create comprehensive customer feature set"""
        # Calculate RFM metrics
        current_date = datetime.now()
        customer_metrics = transaction_data.groupby('customer_id').agg({
            'transaction_date': ['max', 'count'],
            'total_amount': ['sum', 'mean'],
            'quantity': 'sum',
            'discount_amount': 'sum'
        }).reset_index()
        
        # Flatten column names
        customer_metrics.columns = [
            'customer_id', 'last_purchase_date', 'frequency',
            'monetary_total', 'monetary_avg', 'total_items', 'total_discounts'
        ]
        
        # Calculate recency
        customer_metrics['recency'] = (
            current_date - pd.to_datetime(customer_metrics['last_purchase_date'])
        ).dt.days
        
        # Add customer demographics
        customer_features = customer_metrics.merge(
            customer_data[['customer_id', 'age', 'gender', 'zip_code', 'signup_date']], 
            on='customer_id'
        )
        
        # Calculate additional features
        customer_features['customer_lifetime'] = (
            current_date - pd.to_datetime(customer_features['signup_date'])
        ).dt.days
        
        customer_features['avg_days_between_purchases'] = (
            customer_features['customer_lifetime'] / customer_features['frequency']
        )
        
        customer_features['discount_affinity'] = (
            customer_features['total_discounts'] / customer_features['monetary_total']
        )
        
        # Category preferences
        category_purchases = transaction_data.groupby(['customer_id', 'category']).agg({
            'total_amount': 'sum'
        }).reset_index()
        
        category_pivot = category_purchases.pivot(index='customer_id', columns='category', values='total_amount').fillna(0)
        category_pivot.columns = [f'category_{col}_spend' for col in category_pivot.columns]
        
        customer_features = customer_features.merge(category_pivot, left_on='customer_id', right_index=True, how='left')
        
        return customer_features
    
    def segment_customers(self, customer_features):
        """Perform customer segmentation using advanced clustering"""
        # Select features for clustering
        clustering_features = [
            'recency', 'frequency', 'monetary_total', 'monetary_avg',
            'avg_days_between_purchases', 'discount_affinity'
        ]
        
        # Add category spending features
        category_features = [col for col in customer_features.columns if col.startswith('category_')]
        clustering_features.extend(category_features)
        
        # Prepare data for clustering
        X = customer_features[clustering_features].fillna(0)
        X_scaled = self.scaler.fit_transform(X)
        
        # Perform clustering
        self.customer_segmenter.fit(X_scaled)
        customer_features['segment'] = self.customer_segmenter.labels_
        
        # Create segment profiles
        self.segment_profiles = {}
        for segment in range(5):
            segment_data = customer_features[customer_features['segment'] == segment]
            
            self.segment_profiles[segment] = {
                'size': len(segment_data),
                'avg_recency': segment_data['recency'].mean(),
                'avg_frequency': segment_data['frequency'].mean(),
                'avg_monetary': segment_data['monetary_total'].mean(),
                'avg_order_value': segment_data['monetary_avg'].mean(),
                'discount_affinity': segment_data['discount_affinity'].mean(),
                'top_categories': self.get_top_categories(segment_data, category_features),
                'segment_name': self.name_segment(segment_data)
            }
        
        return customer_features
    
    def name_segment(self, segment_data):
        """Assign meaningful names to customer segments"""
        avg_recency = segment_data['recency'].mean()
        avg_frequency = segment_data['frequency'].mean()
        avg_monetary = segment_data['monetary_total'].mean()
        
        if avg_recency < 30 and avg_frequency > 10 and avg_monetary > 1000:
            return 'VIP Champions'
        elif avg_recency < 60 and avg_frequency > 5:
            return 'Loyal Customers'
        elif avg_recency < 90 and avg_monetary > 500:
            return 'Potential Loyalists'
        elif avg_recency > 180:
            return 'At Risk'
        elif avg_frequency < 3:
            return 'New Customers'
        else:
            return 'Regular Customers'
    
    def generate_segment_strategies(self, segment_id):
        """Generate targeted marketing strategies for each segment"""
        profile = self.segment_profiles[segment_id]
        strategies = []
        
        if profile['segment_name'] == 'VIP Champions':
            strategies = [
                'Exclusive early access to new products',
                'Premium customer service line',
                'Personalized product recommendations',
                'VIP events and experiences'
            ]
        elif profile['segment_name'] == 'At Risk':
            strategies = [
                'Win-back email campaigns with special offers',
                'Personalized discount codes',
                'Survey to understand reasons for absence',
                'Retargeting ads with compelling offers'
            ]
        elif profile['segment_name'] == 'New Customers':
            strategies = [
                'Welcome series email campaigns',
                'First purchase incentives',
                'Product education content',
                'Social media engagement'
            ]
        
        return strategies
