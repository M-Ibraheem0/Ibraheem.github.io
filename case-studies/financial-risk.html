# Advanced feature engineering for credit risk
import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
from datetime import datetime, timedelta

def create_behavioral_features(transaction_data, member_id):
    member_txns = transaction_data[transaction_data['member_id'] == member_id]
    
    features = {}
    
    # Cash flow stability metrics
    monthly_balances = member_txns.groupby('month')['balance'].mean()
    features['balance_volatility'] = monthly_balances.std() / monthly_balances.mean()
    features['balance_trend'] = np.polyfit(range(len(monthly_balances)), monthly_balances, 1)[0]
    
    # Payment behavior patterns
    payment_txns = member_txns[member_txns['type'] == 'payment']
    features['avg_payment_amount'] = payment_txns['amount'].mean()
    features['payment_consistency'] = 1 - (payment_txns['amount'].std() / payment_txns['amount'].mean())
    
    # Overdraft and fee patterns
    overdraft_txns = member_txns[member_txns['type'] == 'overdraft']
    features['overdraft_frequency'] = len(overdraft_txns) / len(member_txns)
    features['avg_overdraft_amount'] = overdraft_txns['amount'].mean() if len(overdraft_txns) > 0 else 0
    
    # Seasonal patterns
    features['income_seasonality'] = calculate_seasonality(member_txns, 'deposit')
    features['spending_seasonality'] = calculate_seasonality(member_txns, 'withdrawal')
    
    return features

def calculate_seasonality(txns, txn_type):
    monthly_amounts = txns[txns['type'] == txn_type].groupby('month')['amount'].sum()
    return monthly_amounts.std() / monthly_amounts.mean() if len(monthly_amounts) > 0 else 0

# Ensemble model for credit risk prediction
from xgboost import XGBClassifier
from sklearn.ensemble import RandomForestClassifier
from sklearn.neural_network import MLPClassifier
from sklearn.ensemble import VotingClassifier
from sklearn.model_selection import cross_val_score

class CreditRiskEnsemble:
    def __init__(self):
        # Individual models with optimized hyperparameters
        self.xgb_model = XGBClassifier(
            n_estimators=200,
            max_depth=6,
            learning_rate=0.1,
            subsample=0.8,
            random_state=42
        )
        
        self.rf_model = RandomForestClassifier(
            n_estimators=150,
            max_depth=8,
            min_samples_split=5,
            random_state=42
        )
        
        self.nn_model = MLPClassifier(
            hidden_layer_sizes=(100, 50),
            activation='relu',
            solver='adam',
            random_state=42
        )
        
        # Ensemble with weighted voting
        self.ensemble = VotingClassifier(
            estimators=[
                ('xgb', self.xgb_model),
                ('rf', self.rf_model),
                ('nn', self.nn_model)
            ],
            voting='soft',
            weights=[0.4, 0.35, 0.25]  # XGB gets highest weight
        )
    
    def train(self, X_train, y_train):
        self.ensemble.fit(X_train, y_train)
        
        # Calculate feature importance
        self.feature_importance = self.calculate_ensemble_importance(X_train)
        
        return self
    
    def predict_risk_score(self, X):
        # Return probability of default (0-1 scale)
        return self.ensemble.predict_proba(X)[:, 1]
    
    def get_risk_category(self, risk_score):
        if risk_score < 0.2:
            return 'Low Risk', 'AUTO_APPROVE'
        elif risk_score < 0.4:
            return 'Medium Risk', 'MANUAL_REVIEW'
        elif risk_score < 0.7:
            return 'High Risk', 'ADDITIONAL_COLLATERAL'
        else:
            return 'Very High Risk', 'DECLINE'
