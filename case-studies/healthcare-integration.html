# Healthcare data integration with privacy protection
import pandas as pd
import numpy as np
from cryptography.fernet import Fernet
from datetime import datetime, timedelta
import hashlib

class HealthcareDataIntegrator:
    def __init__(self, encryption_key):
        self.cipher = Fernet(encryption_key)
        self.patient_mapping = {}
        
    def create_master_patient_index(self, emr_data, lab_data, pharmacy_data):
        """Create unified patient identifiers across systems"""
        master_index = []
        
        for system_name, data in [('EMR', emr_data), ('LAB', lab_data), ('PHARMACY', pharmacy_data)]:
            for record in data:
                # Create deterministic hash for patient matching
                patient_key = self.create_patient_hash(
                    record['first_name'],
                    record['last_name'], 
                    record['dob'],
                    record.get('ssn_last4', '')
                )
                
                master_index.append({
                    'master_patient_id': patient_key,
                    'system': system_name,
                    'system_patient_id': record['patient_id'],
                    'confidence_score': self.calculate_match_confidence(record)
                })
                
        return pd.DataFrame(master_index)
    
    def create_patient_hash(self, first_name, last_name, dob, ssn_last4):
        """Create secure, deterministic patient identifier"""
        # Normalize and concatenate identifying information
        identifier = f"{first_name.lower().strip()}{last_name.lower().strip()}{dob}{ssn_last4}"
        
        # Create SHA-256 hash
        return hashlib.sha256(identifier.encode()).hexdigest()[:16]
    
    def integrate_clinical_data(self, master_patient_id):
        """Aggregate all clinical data for a patient"""
        patient_record = {
            'master_patient_id': master_patient_id,
            'demographics': self.get_demographics(master_patient_id),
            'diagnoses': self.get_diagnoses(master_patient_id),
            'medications': self.get_medications(master_patient_id),
            'lab_results': self.get_lab_results(master_patient_id),
            'vital_signs': self.get_vital_signs(master_patient_id),
            'procedures': self.get_procedures(master_patient_id),
            'admissions': self.get_admission_history(master_patient_id)
        }
        
        # Calculate derived clinical features
        patient_record['clinical_features'] = self.calculate_clinical_features(patient_record)
        
        return patient_record
    
    def calculate_clinical_features(self, patient_record):
        """Calculate ML features from clinical data"""
        features = {}
        
        # Comorbidity scoring
        diagnoses = patient_record['diagnoses']
        features['charlson_score'] = self.calculate_charlson_index(diagnoses)
        features['diagnosis_count'] = len(diagnoses)
        features['chronic_conditions'] = len([d for d in diagnoses if d['chronic'] == True])
        
        # Medication complexity
        medications = patient_record['medications']
        features['medication_count'] = len(medications)
        features['high_risk_meds'] = len([m for m in medications if m['high_risk'] == True])
        features['polypharmacy_risk'] = 1 if len(medications) > 5 else 0
        
        # Lab value trends
        labs = patient_record['lab_results']
        if labs:
            features['abnormal_lab_ratio'] = len([l for l in labs if l['abnormal'] == True]) / len(labs)
            features['critical_lab_count'] = len([l for l in labs if l['critical'] == True])
        
        # Utilization patterns
        admissions = patient_record['admissions']
        features['prior_admissions_30d'] = len([a for a in admissions 
                                              if (datetime.now() - a['admit_date']).days <= 30])
        features['prior_admissions_1yr'] = len([a for a in admissions 
                                              if (datetime.now() - a['admit_date']).days <= 365])
        
        return features

# Clinical ML model for readmission prediction
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score
from sklearn.metrics import roc_auc_score, classification_report
import shap

class ReadmissionPredictor:
    def __init__(self):
        # Ensemble of interpretable models for clinical use
        self.models = {
            'logistic': LogisticRegression(random_state=42, max_iter=1000),
            'random_forest': RandomForestClassifier(n_estimators=100, random_state=42),
            'gradient_boost': GradientBoostingClassifier(random_state=42)
        }
        self.feature_names = []
        self.explainer = None
        
    def prepare_features(self, patient_data):
        """Convert clinical data to ML features"""
        features = []
        
        # Demographics
        features.extend([
            patient_data['age'],
            1 if patient_data['gender'] == 'M' else 0,
            patient_data['insurance_type_encoded']
        ])
        
        # Clinical features
        clinical = patient_data['clinical_features']
        features.extend([
            clinical['charlson_score'],
            clinical['diagnosis_count'],
            clinical['medication_count'],
            clinical['prior_admissions_30d'],
            clinical['prior_admissions_1yr'],
            clinical.get('abnormal_lab_ratio', 0),
            clinical.get('critical_lab_count', 0)
        ])
        
        # Vital signs (latest values)
        vitals = patient_data.get('vital_signs', [])
        if vitals:
            latest_vitals = vitals[-1]
            features.extend([
                latest_vitals.get('systolic_bp', 120),
                latest_vitals.get('heart_rate', 70),
                latest_vitals.get('temperature', 98.6),
                latest_vitals.get('oxygen_sat', 98)
            ])
        else:
            features.extend([120, 70, 98.6, 98])  # Default values
            
        # Social determinants
        features.extend([
            patient_data.get('transportation_barrier', 0),
            patient_data.get('social_support_score', 5),
            patient_data.get('health_literacy_score', 5)
        ])
        
        return np.array(features).reshape(1, -1)
    
    def train_ensemble(self, X_train, y_train):
        """Train ensemble of models"""
        self.feature_names = [
            'age', 'gender_male', 'insurance_type',
            'charlson_score', 'diagnosis_count', 'medication_count',
            'prior_admissions_30d', 'prior_admissions_1yr',
            'abnormal_lab_ratio', 'critical_lab_count',
            'systolic_bp', 'heart_rate', 'temperature', 'oxygen_sat',
            'transportation_barrier', 'social_support', 'health_literacy'
        ]
        
        # Train individual models
        for name, model in self.models.items():
            model.fit(X_train, y_train)
            
        # Setup SHAP explainer for interpretability
        self.explainer = shap.TreeExplainer(self.models['random_forest'])
        
        return self
    
    def predict_readmission_risk(self, patient_data):
        """Predict 30-day readmission risk"""
        features = self.prepare_features(patient_data)
        
        # Get predictions from all models
        predictions = {}
        for name, model in self.models.items():
            pred_proba = model.predict_proba(features)[0][1]
            predictions[name] = pred_proba
            
        # Ensemble prediction (weighted average)
        ensemble_pred = (
            predictions['logistic'] * 0.3 +
            predictions['random_forest'] * 0.4 +
            predictions['gradient_boost'] * 0.3
        )
        
        # Get feature importance explanations
        shap_values = self.explainer.shap_values(features)[1]
        
        return {
            'readmission_probability': ensemble_pred,
            'risk_category': self.categorize_risk(ensemble_pred),
            'key_risk_factors': self.get_top_risk_factors(shap_values),
            'recommendations': self.generate_recommendations(ensemble_pred, patient_data)
        }
    
    def categorize_risk(self, probability):
        """Categorize patient risk level"""
        if probability > 0.7:
            return 'HIGH_RISK'
        elif probability > 0.4:
            return 'MEDIUM_RISK'
        else:
            return 'LOW_RISK'
            
    def generate_recommendations(self, risk_prob, patient_data):
        """Generate clinical recommendations based on risk"""
        recommendations = []
        
        if risk_prob > 0.7:
            recommendations.extend([
                'Consider discharge planning consultation',
                'Schedule follow-up within 48-72 hours',
                'Evaluate medication adherence support',
                'Assess social support and transportation needs'
            ])
        elif risk_prob > 0.4:
            recommendations.extend([
                'Schedule follow-up within 1 week',
                'Review medication list for optimization',
                'Consider telehealth monitoring'
            ])
        else:
            recommendations.append('Standard discharge planning appropriate')
            
        return recommendations
