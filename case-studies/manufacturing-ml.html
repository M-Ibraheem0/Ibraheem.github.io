IoT Data Processing Pipeline

# Real-time IoT data processing for predictive maintenance
import pandas as pd
import numpy as np
from scipy import signal
from sklearn.preprocessing import StandardScaler
from datetime import datetime, timedelta

class IoTDataProcessor:
    def __init__(self):
        self.scaler = StandardScaler()
        self.baseline_metrics = {}
        
    def process_sensor_data(self, sensor_readings):
        """Process raw sensor data into ML-ready features"""
        features = {}
        
        # Vibration analysis
        vibration_data = sensor_readings['vibration']
        features['vibration_rms'] = np.sqrt(np.mean(vibration_data**2))
        features['vibration_peak'] = np.max(np.abs(vibration_data))
        features['vibration_kurtosis'] = self.calculate_kurtosis(vibration_data)
        
        # Frequency domain analysis
        frequencies, psd = signal.welch(vibration_data, fs=1000)
        features['dominant_frequency'] = frequencies[np.argmax(psd)]
        features['frequency_spread'] = np.std(frequencies[psd > np.max(psd) * 0.1])
        
        # Temperature trends
        temp_data = sensor_readings['temperature']
        features['temp_mean'] = np.mean(temp_data)
        features['temp_std'] = np.std(temp_data)
        features['temp_trend'] = self.calculate_trend(temp_data)
        features['temp_rate_of_change'] = np.mean(np.diff(temp_data))
        
        # Pressure analysis
        pressure_data = sensor_readings['pressure']
        features['pressure_mean'] = np.mean(pressure_data)
        features['pressure_volatility'] = np.std(pressure_data) / np.mean(pressure_data)
        features['pressure_spikes'] = len(pressure_data[pressure_data > np.percentile(pressure_data, 95)])
        
        # Power consumption patterns
        power_data = sensor_readings['power']
        features['power_efficiency'] = sensor_readings['output_units'] / np.mean(power_data)
        features['power_stability'] = 1 - (np.std(power_data) / np.mean(power_data))
        
        # Cross-sensor correlations
        features['temp_vibration_corr'] = np.corrcoef(temp_data, vibration_data)[0,1]
        features['pressure_power_corr'] = np.corrcoef(pressure_data, power_data)[0,1]
        
        return features
    
    def calculate_health_score(self, features, equipment_id):
        """Calculate overall equipment health score (0-100)"""
        if equipment_id not in self.baseline_metrics:
            return 85  # Default for new equipment
            
        baseline = self.baseline_metrics[equipment_id]
        health_score = 100
        
        # Deduct points for deviations from baseline
        health_score -= min(20, abs(features['vibration_rms'] - baseline['vibration_rms']) * 10)
        health_score -= min(15, abs(features['temp_mean'] - baseline['temp_mean']) * 0.5)
        health_score -= min(15, abs(features['pressure_mean'] - baseline['pressure_mean']) * 0.1)
        health_score -= min(10, (1 - features['power_efficiency'] / baseline['power_efficiency']) * 50)
        
        return max(0, health_score)

Predictive Maintenance Model

# LSTM-based predictive maintenance model
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout
from sklearn.ensemble import IsolationForest

class PredictiveMaintenanceModel:
    def __init__(self, sequence_length=24):
        self.sequence_length = sequence_length
        self.failure_model = self.build_failure_prediction_model()
        self.anomaly_detector = IsolationForest(contamination=0.1, random_state=42)
        
    def build_failure_prediction_model(self):
        """Build LSTM model for failure prediction"""
        model = Sequential([
            LSTM(64, return_sequences=True, input_shape=(self.sequence_length, 15)),
            Dropout(0.2),
            LSTM(32, return_sequences=False),
            Dropout(0.2),
            Dense(16, activation='relu'),
            Dense(1, activation='sigmoid')  # Probability of failure in next 24 hours
        ])
        
        model.compile(
            optimizer='adam',
            loss='binary_crossentropy',
            metrics=['accuracy', 'precision', 'recall']
        )
        
        return model
    
    def predict_failure_probability(self, sensor_sequence):
        """Predict probability of equipment failure"""
        # Reshape for LSTM input
        X = sensor_sequence.reshape(1, self.sequence_length, -1)
        
        # Get failure probability
        failure_prob = self.failure_model.predict(X)[0][0]
        
        # Get anomaly score
        current_features = sensor_sequence[-1].reshape(1, -1)
        anomaly_score = self.anomaly_detector.decision_function(current_features)[0]
        
        # Combine predictions
        risk_score = (failure_prob * 0.7) + (max(0, -anomaly_score) * 0.3)
        
        return {
            'failure_probability': failure_prob,
            'anomaly_score': anomaly_score,
            'risk_score': risk_score,
            'recommendation': self.get_maintenance_recommendation(risk_score)
        }
    
    def get_maintenance_recommendation(self, risk_score):
        """Generate maintenance recommendations based on risk score"""
        if risk_score > 0.8:
            return {
                'action': 'IMMEDIATE_SHUTDOWN',
                'priority': 'CRITICAL',
                'timeframe': 'Within 2 hours',
                'description': 'High failure risk detected. Schedule immediate inspection.'
            }
        elif risk_score > 0.6:
            return {
                'action': 'SCHEDULE_MAINTENANCE',
                'priority': 'HIGH',
                'timeframe': 'Within 24 hours',
                'description': 'Elevated risk. Schedule preventive maintenance.'
            }
        elif risk_score > 0.4:
            return {
                'action': 'MONITOR_CLOSELY',
                'priority': 'MEDIUM',
                'timeframe': 'Within 72 hours',
                'description': 'Increased monitoring recommended. Check key components.'
            }
        else:
            return {
                'action': 'NORMAL_OPERATION',
                'priority': 'LOW',
                'timeframe': 'Next scheduled maintenance',
                'description': 'Equipment operating within normal parameters.'
            }
